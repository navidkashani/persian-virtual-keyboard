How to Use

ES Module

import './src/persian-keyboard.js';

// Then use in HTML:
// <persian-keyboard></persian-keyboard>

React

import { PersianKeyboard } from 'persian-virtual-keyboard/react';

function App() {
  return <PersianKeyboard
    showEnglishValue
    onPersianChange={(e) => console.log(e.detail.value)}
  />;
}
